# HOW THIS WEBSITE PAGE WAS CREATED

Create a new GitHub repository at:
    https://github.com/kwegner7/pictures 
And this will be exported as:
    http://kwegner7.github.io/pictures/index.html

# Create a new .git folder here
  cd /proj/pictures
  git init
  git add . 
  git commit -m "Distributing Pictures"

# Manually create a new repository at GitHub
   click New Repository, then enter name "pictures", then Create, 
   then copy the link: https://github.com/kwegner7/pictures.git

# Add the new Github repository as a remote using the key:
  git remote add github https://7155aa6db6647b440b4f3a46d534de55988f2020@github.com/kwegner7/pictures.git
  git remote -v
  git push github master

# Create a branch so that GitHub will export and allow web pages
# Declare a branch at the current tree
  git branch gh-pages
# This switches the checkout from master to gh-pages
  git checkout gh-pages
# This sets the push to now push the branch, not the master
  git push -u github gh-pages

# Create a backup repository on USB
# It contains a master and a gh-pages branch.
# It contains a remote branch called "origin"
# The only remote for this repository is /proj/pictures/.git
# and it is known as "origin"
  cd /media/kurt/8GB-BLUE4/git
  git clone --mirror /proj/pictures/.git
  cd pictures.git && git remote -v
      origin  /proj/pictures/.git (fetch)
      origin  /proj/pictures/.git (push)
  
# Back at the origin, git the new backup repository a remote name
# and push to it
  cd /proj/pictures
  git remote add usb /media/kurt/8GB-BLUE4/git/pictures.git 
# This sets push to push from the gh-pages branch of origin
# to the gh-pages branch of usb
  git push -u usb gh-pages
  git diff gh-pages usb/gh-pages

# This simplifies the push commands
  git push -u github gh-pages
  git push -u backup gh-pages
